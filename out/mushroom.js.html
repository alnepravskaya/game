import FieldItem from './fieldItem';
import GM from "./GM";
/**
 * Mushroom item appears as background object on the field
 * @extends FieldItem
 * @class Mushroom
 * */
class Mushroom extends FieldItem {
	constructor(param) {
		super(param);

	}
	/**
	 * collided player with mushroom
	 * @override
	 * @fires 'game:mushroom:collect' - player pick up the mushroom
	 */
	collided(){
		GM.trigger('game:mushroom:collect');
		this.hidden=true;
	}

}


export default Mushroom;
