import Layer from "./layer";
import Mushroom from "./mushroom";
/**
 * Layer with mushroom items
 * @class MushroomLayer
 * @extends Layer
 * */


class MushroomLayer extends Layer {
	constructor() {
		super();
		this.config = {
			speed: 2,
			countCoin: 3
		};
		let last = 0;
		for (var i = 0; i < this.config.countCoin; i++) {
			last += Math.floor(Math.random() * (1500 - 400)) + 400;
			let mushroom = new Mushroom({
				x:last,
				y:Math.round(Math.random()*500),
				speedX:this.config.speed,
				imgName:"live",
				collisable: true
			});
			this.items.push(mushroom);
		}
	}
}

export default MushroomLayer;
